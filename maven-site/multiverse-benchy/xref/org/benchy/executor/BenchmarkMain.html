57      private static Benchmark loadBenchmark(String benchmarkJson) {
58          GsonBuilder builder = new GsonBuilder();
59          builder.registerTypeAdapter(Benchmark.class, new BenchmarkDeserializer());
60          builder.registerTypeAdapter(TestCase.class, new TestCaseDeserializer());
61          Gson gson = builder.create();
62          return gson.fromJson(benchmarkJson, Benchmark.class);
63      }

65      static class BenchmarkDeserializer implements JsonDeserializer {
66  
67          @Override
68          public Object deserialize(JsonElement json, Type type, JsonDeserializationContext context) throws JsonParseException {
69              Benchmark benchmark = new Benchmark();
70  
71              JsonObject obj = (JsonObject) json;
72              benchmark.benchmarkName = obj.get("benchmarkName").getAsString();
73              benchmark.driverClass = obj.get("driverClass").getAsString();
74  
75              JsonArray testCaseJsonArray = obj.get("testcases").getAsJsonArray();
76              for (JsonElement element : testCaseJsonArray) {
77                  JsonObject o = (JsonObject) element;
78                  TestCase testCase = context.deserialize(o, TestCase.class);
79                  benchmark.testCases.add(testCase);
80              }
81  
82              return benchmark;
83          }
84      }

86      static class TestCaseDeserializer implements JsonDeserializer {
87          @Override
88          public Object deserialize(JsonElement json, Type type, JsonDeserializationContext jsonDeserializationContext) throws JsonParseException {
89              JsonObject obj = (JsonObject) json;
90  
91              TestCase testCase = new TestCase();
92              for (Map.Entry<String, JsonElement> s : obj.entrySet()) {
93                  testCase.setProperty(s.getKey(), s.getValue().getAsString());
94              }

27      private static void ensureExistingDirectory(File dir) {
28          if (dir == null) {
29              throw new NullPointerException();
30          }
31  
32          if (!dir.isDirectory()) {
33              if (!dir.mkdirs()) {
34                  String msg = format("Unable to create directory %s", dir);
35                  throw new RuntimeException(msg);
36              }
37          }

62      @Override
63      public BenchmarkResult load(Date date, String benchmarkName) {
64          if (date == null || benchmarkName == null) {
65              throw new NullPointerException();
66          }
67  
68          File benchmarkDirectory = getBenchmarkDirectory(benchmarkName);
69  
70          File todaysMeasurementsDir = new File(
71                  benchmarkDirectory,
72                  toDateSeperator(date));
73  
74          List<TestCaseResult> caseResults = new LinkedList<TestCaseResult>();
75          if (todaysMeasurementsDir.isDirectory()) {
76              File resultsDir = findLastRun(todaysMeasurementsDir);
77  
78              if (resultsDir.isDirectory()) {
79                  for (File file : resultsDir.listFiles(new ResultFileFilter())) {
80  
81                      Properties properties = new Properties();
82                      try {
83                          properties.load(new FileReader(file));
84                          TestCaseResult caseResult = new TestCaseResult(properties);
85                          caseResults.add(caseResult);
86                      } catch (IOException e) {
87                          throw new RuntimeException(e);
88                      }
89                  }
90              }
91          }
92          return new BenchmarkResult(benchmarkName, caseResults);
93      }

95      public void store(BenchmarkResult benchmarkResult) {
96          if (benchmarkResult == null) {
97              throw new NullPointerException();
98          }
99  
100         File targetDir = createTargetDir(benchmarkResult.getBenchmarkName());
101 
102         int k = 1;
103         for (TestCaseResult result : benchmarkResult.getTestCaseResultList()) {
104             File outputFile = new File(targetDir, k + ".txt");
105             writeOutputToFile(result, outputFile);
106             k++;
107         }
108     }

110     private void writeOutputToFile(TestCaseResult testCaseResult, File target) {
111         try {
112             Writer output = new BufferedWriter(new FileWriter(target));
113             try {
114                 testCaseResult.getProperties().store(output, "");
115             } finally {
116                 output.close();
117             }
118         } catch (IOException ex) {
119             ex.printStackTrace();
120         }
121     }

123     private File createTargetDir(String benchmarkName) {
124         File benchmarkDirectory = getBenchmarkDirectory(benchmarkName);
125 
126         File todaysMeasurementsDir = new File(
127                 benchmarkDirectory,
128                 toDateSeperator(new Date()));
129 
130         ensureExistingDirectory(todaysMeasurementsDir);
131         return createRunDir(todaysMeasurementsDir);
132     }

134     /***
135      * Creates a directory that points to all files from a run of a single Benchmark.
136      */
137     private File createRunDir(File todaysBenchmarksDir) {
138         int max = 0;
139         for (File file : todaysBenchmarksDir.listFiles()) {
140             try {
141                 String name = file.getName();
142                 int value = Integer.parseInt(name);
143                 if (value > max) {
144                     max = value;
145                 }
146             } catch (NumberFormatException ex) {
147                 //ignore, go to the next file.
148             }
149         }
150 
151         File runDir = new File(todaysBenchmarksDir, "" + (max + 1));
152         ensureExistingDirectory(runDir);
153         return runDir;
154     }

156     private File findLastRun(File todaysBenchmarksDir) {
157         int max = -1;
158         File result = null;
159         for (File file : todaysBenchmarksDir.listFiles()) {
160             try {
161                 String name = file.getName();
162                 int value = Integer.parseInt(name);
163                 if (value > max) {
164                     max = value;
165                     result = file;
166                 }
167             } catch (NumberFormatException ex) {
168                 //ignore, go to the next file.
169             }
170         }
171 
172         return result;
173     }

175     private File getBenchmarkDirectory(String benchmarkName) {
176         File result = rootDir;
177         StringTokenizer tokenizer = new StringTokenizer(benchmarkName);
178         while (tokenizer.hasMoreElements()) {
179             result = new File(result, tokenizer.nextToken());
180         }
181         return result;
182     }

184     private String toDateSeperator(Date date) {
185         SimpleDateFormat format = new SimpleDateFormat("y-M-d");
186         return format.format(date);

14  /***
15   * An abstract {@link Transaction} implementation that contains most of the plumbing logic. Extend
16   * this and prevent duplicate logic.

21   * the constructor of the abstracttransaction because properties in the subclass perhaps are not set.

69      @Override
70      public void deferredExecute(Runnable task) {
71          switch (status) {
72              case active:
73                  if (task == null) {
74                      throw new NullPointerException();
75                  }
76                  if (deferredTasks == null) {
77                      deferredTasks = new LinkedList<Runnable>();
78                  }
79                  deferredTasks.add(task);
80                  break;
81              case committed: {
82                  String msg = format("Can't add deferredExecute task on already committed transaction '%s'", familyName);
83                  throw new DeadTransactionException(msg);
84              }
85              case aborted: {
86                  String msg = format("Can't add deferredExecute task on already aborted transaction '%s'", familyName);
87                  throw new DeadTransactionException(msg);
88              }
89              default:
90                  throw new RuntimeException();

208     public long commit() {
209         switch (status) {
210             case active:
211                 boolean abort = true;
212                 try {
213                     commitVersion = onCommit();
214                     abort = false;
215                     return commitVersion;
216                 } finally {
217                     if (abort) {
218                         doAbort();
219                     } else {
220                         status = TransactionStatus.committed;
221                         executeDeferredTasks();
222                     }
223                 }
224             case committed:
225                 return commitVersion;

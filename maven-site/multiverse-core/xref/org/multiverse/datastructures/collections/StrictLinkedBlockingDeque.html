198     private Node<E> getNode(int index) {
199         Node<E> node;
200         if (index < size / 2) {
201             node = head;
202             for (int k = 0; k < index; k++) {
203                 node = node.next;
204             }
205         } else {
206             node = tail;
207             for (int k = size - 1; k > index; k--) {

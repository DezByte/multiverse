6   /***
7    * A BlockingStack interface similar to the {@link java.util.concurrent.BlockingQueue} and the
8    * {@link java.util.concurrent.BlockingDeque} interface.
9    * <p/>
10   * The java.util.Stack class has no interface and there is no explicit BlockingStack interface (there is a
11   * {@link java.util.concurrent.BlockingDeque} that can be used as a Stack interface. That is why one was created for the Multiverse
12   * project so that different stack implementations can be used without changing (too much) code.
13   * <p/>
14   * <p>Memory consistency effects: As with other concurrent collections, actions in a thread prior to placing an object into a
15   * {@code BlockingStack}
16   * <a href="package-summary.html#MemoryVisibility"><i>happen-before</i></a> actions subsequent to the access or removal
17   * of that element from the {@code BlockingStack} in another thread.
18   *
19   * @param <E>
20   */
21  public interface BlockingStack<E> extends Collection<E> {
22  
23      /***
24       * Returns the number of additional elements that this stack can ideally
25       * (in the absence of memory or resource constraints) accept without
26       * blocking, or <tt>Integer.MAX_VALUE</tt> if there is no intrinsic
27       * limit.
28       * <p/>
29       * <p>Note that you <em>cannot</em> always tell if an attempt to insert
30       * an element will succeed by inspecting <tt>remainingCapacity</tt>
31       * because it may be the case that another thread is about to
32       * insert or remove an element.
33       *
34       * @return the remaining capacity
35       */
36      int getRemainingCapacity();
37  
38      /***
39       * Retrieves, but does not remove the last pushed item of this Stack.
40       *
41       * @return the the last pushed item of the stack, or null if the stack is empty.
42       */
43      E peek();
44  
45      /***
46       * Pops an item of the stack. Waiting if necessary for an item to be placed.
47       *
48       * @return the popped item.
49       */
50      E pop();
51  
52      /***
53       * Retrieves and removes most recently added item.
54       *
55       * @return the first element of this stack, or <tt>null</tt> if this stack is empty
56       */
57      E poll();
58  
59      /***
60       * Pushes an item on the stack  or fails with an IllegalStateException if the item can't
61       * be placed due to capacity restrictions.
62       *
63       * @param item the item to push.
64       * @throws NullPointerException  if item is null.
65       * @throws IllegalStateException if the element cannot be added at this
66       *                               time stack to capacity restrictions
67       */
68      void push(E item);
69  
70      /***
71       * Puts an item on the stack. Waiting if necessary for space to become available.
72       *
73       * @param item the item to add to this stack.
74       * @throws NullPointerException if item is null.
75       * @throws InterruptedException if the thread was interrupted during the execution
76       */
77      void put(E item) throws InterruptedException;
78  
79      /***
80       * Offers an item to the stack.
81       *
82       * @param item the item to add to this stack
83       * @return true if the item was added successfully, false otherwise.
84       * @throws NullPointerException if item is null.
85       */
86      boolean offer(E item);
87  
88      boolean offer(E item, long timeout, TimeUnit unit) throws InterruptedException;
89  
90      /***
91       * Removes all available elements from this stack and adds them to the given collection.  This operation may be more
92       * efficient than repeatedly polling this stack.  A failure encountered while attempting to add elements to
93       * collection <tt>c</tt> may result in elements being in neither, either or both collections when the associated exception is
94       * thrown.  Attempts to drain a stack to itself result in <tt>IllegalArgumentException</tt>.
95       *
96       * @param c the collection to transfer elements into
97       * @return the number of elements transferred
98       * @throws UnsupportedOperationException if addition of elements
99       *                                       is not supported by the specified collection
100      * @throws ClassCastException            if the class of an element of this stack
101      *                                       prevents it from being added to the specified collection
102      * @throws NullPointerException          if the specified collection is null
103      * @throws IllegalArgumentException      if the specified collection is this
104      *                                       stack, or some property of an element of this stack prevents
105      *                                       it from being added to the specified collection
106      */
107     int drainTo(Collection<? super E> c);
108 
109     /***
110      * Removes at most the given number of available elements from this stack and adds them to the given collection.  A failure
111      * encountered while attempting to add elements to collection <tt>c</tt> may result in elements being in neither,
112      * either or both collections when the associated exception is thrown.  Attempts to stack a stack to itself result in
113      * <tt>IllegalArgumentException</tt>.
114      *
115      * @param c           the collection to transfer elements into
116      * @param maxElements the maximum number of elements to transfer
117      * @return the number of elements transferred
118      * @throws UnsupportedOperationException if addition of elements
119      *                                       is not supported by the specified collection
120      * @throws ClassCastException            if the class of an element of this stack
121      *                                       prevents it from being added to the specified collection
122      * @throws NullPointerException          if the specified collection is null
123      * @throws IllegalArgumentException      if the specified collection is this
124      *                                       stack, or some property of an element of this stack prevents
125      *                                       it from being added to the specified collection
126      */
127     int drainTo(Collection<? super E> c, int maxElements);

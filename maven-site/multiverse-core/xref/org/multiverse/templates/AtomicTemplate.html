17  /***
18   * A Template that handles the boilerplate code for transactions. A transaction will be placed if none is available
19   * around a section and if all goes right, commits at the end.
20   * <p/>
21   * example:
22   * <pre>
23   * new AtomicTemplate(){
24   *    Object execute(Transaction t){
25   *        queue.push(1);
26   *        return null;
27   *    }
28   * }.execute();
29   * </pre>
30   * <p/>
31   * It could also be that the transaction is retried (e.g. caused by optimistic locking failures). This is also a task
32   * for template. In the future this retry behavior will be customizable.
33   * <p/>
34   * If a transaction already is available on the ThreadLocalTransaction, no new transaction is started and essentially
35   * the whole AtomicTemplate is ignored.
36   * <p/>
37   * If no transaction is available on the ThreadLocalTransaction, a new one will be created and used during the execution
38   * of the AtomicTemplate and will be removed once the AtomicTemplate finishes.
39   * <p/>
40   * All uncaught throwable's lead to a rollback of the transaction.
41   * <p/>
42   * AtomicTemplates are not thread-safe to use.
43   * <p/>
44   * AtomicTemplates can completely work without threadlocals. See the {@link AtomicTemplate#AtomicTemplate(org.multiverse.api.Stm
45   * ,String, boolean, boolean, int)} for more information.
46   *
47   * @author Peter Veentjer
48   */
49  public abstract class AtomicTemplate<E> {

268     /***
269      * Executes the Template and rethrows the checked exception instead of wrapping it in a InvisibleCheckedException.

272      * @throws Exception               the Exception thrown inside the {@link #execute(org.multiverse.api.Transaction)}
273      *                                 method.
274      * @throws AbortedException        if the exception was explicitly aborted.
275      * @throws TooManyRetriesException if the template retried the transaction too many times. The cause of the last
276      *                                 failure (also an exception) is included as cause. So you have some idea where to
277      *                                 look for problems
278      */
279     public final E executeChecked() throws Exception {
280         Transaction t = getTransaction();
281         if (noUsableTransaction(t)) {
282             return executeAtomic();
283         } else {
284             return execute(t);
285         }

288     private E executeAtomic() throws Exception {
289         onInit();
290         Transaction t = startTransaction();
291         setTransaction(t);
292         try {
293             attemptCount = 1;
294             Throwable lastFailureCause = null;
295             while (attemptCount - 1 <= retryCount) {
296                 boolean abort = true;
297                 boolean reset = false;
298                 try {
299                     postStart(t);
300                     E result = execute(t);
301                     if (t.getStatus().equals(TransactionStatus.aborted)) {
302                         String msg = format("Transaction with familyName '%s' is aborted", t.getFamilyName());
303                         throw new AbortedException(msg);
304                     }
305                     preCommit(t);
306                     long commitVersion = t.commit();
307                     abort = false;
308                     reset = false;
309                     postCommit(commitVersion);
310                     return result;
311                 } catch (RetryError e) {
312                     t.abortAndWaitForRetry();
313                     //since the abort is already done, no need to do it again.
314                     abort = false;
315                 } catch (Throwable throwable) {
316                     if (throwable instanceof RecoverableThrowable) {
317                         lastFailureCause = throwable;
318                         reset = true;
319                     } else {
320                         rethrow(throwable);
321                     }
322                 } finally {
323                     if (abort) {
324                         t.abort();
325                         postAbort(t);
326                         if (reset) {
327                             t.restart();
328                         }
329                     }
330                 }
331                 attemptCount++;
332             }
333 
334             String msg = format("Too many retries, maximum number of retries = %s", retryCount);
335             throw new TooManyRetriesException(msg, lastFailureCause);
336         } finally {
337             setTransaction(null);
338         }
339     }

341     private static void rethrow(Throwable ex) throws Exception {
342         if (ex instanceof Exception) {
343             throw (Exception) ex;
344         } else if (ex instanceof Error) {
345             throw (Error) ex;
346         } else {
347             throw new PanicError("Unthrowable throwable", ex);
348         }
349     }
